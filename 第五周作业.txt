#建如下两张表 
create table dept 
( 
deptno int(11) primary key,  #部门号
dname varchar(10)         #部门名
); 
create table emp 
( 
empno int(11) primary key,  #员工工号
ename varchar(10),        #员工姓名
job varchar(10),           #员工岗位
mgr varchar(10),          #员工上司
sal varchar(10),           #员工薪水
deptno int(11),           #所在部门号
foreign key(deptno) references dept(deptno) 
);
